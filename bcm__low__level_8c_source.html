93 bool init_bsc_i2c_slv(uint8_t i2c_addr)
94 {
95  if (!bsc) {

97  return false;
98  }
99 
100  // Alternative function 3 is for BSC
101  gpio_set_mode(GPIO_SDA, GPIO_FUN_ALT3);
102  gpio_set_mode(GPIO_SCL, GPIO_FUN_ALT3);
103 
104  bsc[BSC_CR] = CR_BRK; // First reset everything
105  bsc[BSC_RSR] = 0;
106  bsc[BSC_IMSC] = 0xf;
107  bsc[BSC_ICR] = 0xf;
108 
109  // Shift addr right one to get 7 bit addr without RW bit.
110  bsc[BSC_SLV] = (i2c_addr>>1);
111  bsc[BSC_CR] = CR_TXE | CR_RXE | CR_I2C | CR_EN;
112 
113  return true;

116 bool bcm_set_gpio_out(int gpio, enum gpio_state state)
117 {
118  if (!gpio_reg) {
119  fprintf(stderr, TAG ": init_bcm_reg_mem() has not been called\n");
120  return false;
121  }
122  if (gpio < 0 || gpio >= GPIO_COUNT) {
123  fprintf(stderr, TAG ": Invalid GPIO: %d\n", gpio);
124  return false;
125  }
126  switch(state) {
127  case GPIO_STATE_FLOAT:
128  // Using GPIO input as a way to get the gpio to float.
129  // We have to do this because the Raspberry Pi lacks an open drain
130  // mode, which is what we would really want for this pin.
131  gpio_set_mode(gpio, GPIO_FUN_IN);
132  return true;
133  case GPIO_STATE_LOW:
134  gpio_reg[GPCLR0] |= ((uint32_t)1) << gpio;
135  gpio_set_mode(gpio, GPIO_FUN_OUT);
136  return true;
137  case GPIO_STATE_HIGH:
138  gpio_reg[GPSET0] |= ((uint32_t)1) << gpio;
139  gpio_set_mode(gpio, GPIO_FUN_OUT);
140  return true;
141  default:
142  fprintf(stderr, TAG ": Invalid GPIO state: %d\n", state);
143  return false;
144  }
145 }

156 int bsc_i2c_read_poll(uint8_t * buf, size_t len)
157 {
158  pthread_testcancel();
159  if (len == 0 || buf == NULL) {
160  return 0;
161  }
162 
163  size_t read = 0;
164 
165  // Loop as long as:
166  // 1. We have room to receive data.
167  // 2. The RX fifo has data.
168  // 3. This thread has not been canceled
169  for (; len && !RX_EMPTY(); len--) {
170  pthread_testcancel();
171  if (bsc[BSC_RSR] & RSR_OE) {
172  // We overflowed. :-(
173  fprintf(stderr, TAG ": Overflow!\n");
174  bsc[BSC_RSR] &= ~RSR_OE; // Clear the overflow error
175  }
176  buf[read] = bsc[BSC_DR] & 0xFF;
177  read++;
178  }
179 
180  return read;
181 }

183 int bsc_i2c_write(tx_callback cb, uint16_t addr)
184 {
185  pthread_testcancel();
186  int offset = 0;
187 
188  // Keep replying as long as the master is not writing to us.
189  while (RX_EMPTY()) {
190  pthread_testcancel();
191  // Keep the TX FIFO full
192  while ( !(bsc[BSC_FR] & FR_TXFF)) {
193  pthread_testcancel();
194  // Check for underflows
195  if (bsc[BSC_RSR] & RSR_UE) {
196  // We had an underrun happen. :-(
197  fprintf(stderr, TAG ": Underrun!\n");
198  bsc[BSC_RSR] &= ~RSR_UE; // Clear the underrun error
199  }
200  uint8_t byte;
201  if (cb(addr++, &byte)) {
202  bsc[BSC_DR] = byte;
203  offset++;
204  } else {
205  // We have used up all the data this callback has.
206  break;
207  }
208  }
209  usleep(WRITE_USLEEP_INTERVAL);
210  }
211 
212  // Return value is how many bytes we put in the TX FIFO minus the number of
213  // bytes that are left in it minus an additional byte which got sucked off
214  // the FIFO, ready to be sent, but never was sent.
215  int ret = offset - GET_FR_TXFLEVEL() - 1;
216 
217  // We need to get the TX FIFO clear, otherwise the next time the master
218  // does a read, it will get the unread leftovers from this read.
219  //
220  // The method here is a bit hacky, but it is the only one I could find. I
221  // was unable to get CR_BRK to work at all. It simply would not clear the
222  // fifo. As far as I could see, it does nothing. I found numerous places
223  // online where people reported the same issue.
224  //
225  // The method is simple, whenever TX is disabled and re-enabled, it drops
226  // the current TX byte and pops the next one out of the FIFO, so we can
227  // put this in a while loop until the TX FIFO is empty (TXFE) and then
228  // do it one more time to drop the last TX byte.
229  //
230  // Note: this behavior is undocumented as far as I can tell. The BCM2837
231  // ARM Peripherals specification document doesn't mention it. However,
232  // that spec is generally known to contain errors and omissions.
233  while (!(bsc[BSC_FR] & FR_TXFE)) {
234  bsc[BSC_CR] &= ~CR_TXE;
235  bsc[BSC_CR] |= CR_TXE;
236  }
237  bsc[BSC_CR] &= ~CR_TXE;
238  bsc[BSC_CR] |= CR_TXE;
239 
240  // When this software is first getting running, I have seen some
241  // instability. This is just a sanity check.
242  return (ret < 0) ? 0 : ret;
243 }
