77 static void gpio_set_mode(uint32_t gpio, uint32_t mode)
78 {

86 bool init_bsc_i2c_slv(uint8_t i2c_addr)
87 {
88  if (!bsc) {
89  fprintf(stderr, TAG ": init_bcm_reg_mem() has not been called\n");
90  return false;
91  }
92 
93  // Alternative function 3 is for BSC
94  gpio_set_mode(GPIO_SDA, GPIO_FUN_ALT3);
95  gpio_set_mode(GPIO_SCL, GPIO_FUN_ALT3);
96 
97  bsc[BSC_CR] = CR_BRK; // First reset everything
98  bsc[BSC_RSR] = 0;
99  bsc[BSC_IMSC] = 0xf;
100  bsc[BSC_ICR] = 0xf;
101 
102  // Shift addr right one to get 7 bit addr without RW bit.
103  bsc[BSC_SLV] = (i2c_addr>>1);
104  bsc[BSC_CR] = CR_TXE | CR_RXE | CR_I2C | CR_EN;
105 
106  return true;
107 }

109 bool bcm_set_gpio_out(int gpio, enum gpio_state state)
110 {
111  if (!gpio_reg) {
112  fprintf(stderr, TAG ": init_bcm_reg_mem() has not been called\n");
113  return false;
114  }
115  if (gpio < 0 || gpio >= GPIO_COUNT) {
116  fprintf(stderr, TAG ": Invalid GPIO: %d\n", gpio);
117  return false;
118  }
119  switch(state) {
120  case GPIO_STATE_FLOAT:
121  // Using GPIO input as a way to get the gpio to float.
122  // We have to do this because the Raspberry Pi lacks an open drain
123  // mode, which is what we would really want for this pin.
124  gpio_set_mode(gpio, GPIO_FUN_IN);
125  return true;
126  case GPIO_STATE_LOW:
127  gpio_reg[GPCLR0] |= ((uint32_t)1) << gpio;
128  gpio_set_mode(gpio, GPIO_FUN_OUT);
129  return true;
130  case GPIO_STATE_HIGH:
131  gpio_reg[GPSET0] |= ((uint32_t)1) << gpio;
132  gpio_set_mode(gpio, GPIO_FUN_OUT);
133  return true;
134  default:
135  fprintf(stderr, TAG ": Invalid GPIO state: %d\n", state);
136  return false;
137  }
138 }

149 int bsc_i2c_read_poll(uint8_t * buf, size_t len)
150 {
151  pthread_testcancel();
152  if (len == 0 || buf == NULL) {
153  return 0;
154  }
155 
156  size_t read = 0;
157 
158  // Loop as long as:
159  // 1. We have room to receive data.
160  // 2. The RX fifo has data.
161  // 3. This thread has not been canceled
162  for (; len && !RX_EMPTY(); len--) {
163  pthread_testcancel();
164  if (bsc[BSC_RSR] & RSR_OE) {
165  // We overflowed. :-(
166  fprintf(stderr, TAG ": Overflow!\n");
167  bsc[BSC_RSR] &= ~RSR_OE; // Clear the overflow error
168  }
169  buf[read] = bsc[BSC_DR] & 0xFF;
170  read++;
171  }
172 
173  return read;
174 }

176 int bsc_i2c_write(tx_callback cb, uint16_t addr)
177 {
178  pthread_testcancel();
179  int offset = 0;
180 
181  // Keep replying as long as the master is not writing to us.
182  while (RX_EMPTY()) {
183  pthread_testcancel();
184  // Keep the TX FIFO full
185  while ( !(bsc[BSC_FR] & FR_TXFF)) {
186  pthread_testcancel();
187  // Check for underflows
188  if (bsc[BSC_RSR] & RSR_UE) {
189  // We had an underrun happen. :-(
190  fprintf(stderr, TAG ": Underrun!\n");
191  bsc[BSC_RSR] &= ~RSR_UE; // Clear the underrun error
192  }
193  uint8_t byte;
194  if (cb(addr++, &byte)) {
195  bsc[BSC_DR] = byte;
196  offset++;
197  } else {
198  // We have used up all the data this callback has.
199  break;
200  }
201  }
202  usleep(WRITE_USLEEP_INTERVAL);
203  }
204 
205  // Return value is how many bytes we put in the TX FIFO minus the number of
206  // bytes that are left in it minus an additional byte which got sucked off
207  // the FIFO, ready to be sent, but never was sent.
208  int ret = offset - GET_FR_TXFLEVEL() - 1;
209 
210  // We need to get the TX FIFO clear, otherwise the next time the master
211  // does a read, it will get the unread leftovers from this read.
212  //
213  // The method here is a bit hacky, but it is the only one I could find. I
214  // was unable to get CR_BRK to work at all. It simply would not clear the
215  // fifo. As far as I could see, it does nothing. I found numerous places
216  // online where people reported the same issue.
217  //
218  // The method is simple, whenever TX is disabled and re-enabled, it drops
219  // the current TX byte and pops the next one out of the FIFO, so we can
220  // put this in a while loop until the TX FIFO is empty (TXFE) and then
221  // do it one more time to drop the last TX byte.
222  //
223  // Note: this behavior is undocumented as far as I can tell. The BCM2837
224  // ARM Peripherals specification document doesn't mention it. However,
225  // that spec is generally known to contain errors and omissions.
226  while (!(bsc[BSC_FR] & FR_TXFE)) {
227  bsc[BSC_CR] &= ~CR_TXE;
228  bsc[BSC_CR] |= CR_TXE;
229  }
230  bsc[BSC_CR] &= ~CR_TXE;
231  bsc[BSC_CR] |= CR_TXE;
232 
233  // When this software is first getting running, I have seen some
234  // instability. This is just a sanity check.
235  return (ret < 0) ? 0 : ret;
236 }
